document.addEventListener('DOMContentLoaded', () => { const domainInput = document.getElementById('domain'); const lookupButton = document.getElementById('lookupButton'); const resultsDiv = document.getElementById('results'); // DNS over HTTPS resolver const DOH_ENDPOINT = 'https://cloudflare-dns.com/dns-query'; // Common DNS record types to check const RECORD_TYPES = [ { type: 'A', name: 'A Records (IPv4)' }, { type: 'AAAA', name: 'AAAA Records (IPv6)' }, { type: 'CNAME', name: 'CNAME Records' }, { type: 'MX', name: 'MX Records (Mail)' }, { type: 'TXT', name: 'TXT Records' }, { type: 'NS', name: 'NS Records (Name Servers)' }, { type: 'SOA', name: 'SOA Record (Authority)' }, { type: 'PTR', name: 'PTR Records (Reverse)' }, { type: 'SRV', name: 'SRV Records (Services)' }, { type: 'CAA', name: 'CAA Records (Certificate Authority)' } ]; const checkDNS = async (domain, recordType) => { try { const url = `${DOH_ENDPOINT}?name=${encodeURIComponent(domain)}&type=${recordType}`; const response = await fetch(url, { headers: { 'Accept': 'application/dns-json' } }); if (!response.ok) { throw new Error(`HTTP ${response.status}: ${response.statusText}`); } const data = await response.json(); return data; } catch (error) { return { Status: -1, error: error.message }; } }; const formatRecordData = (record, recordType) => { switch (recordType) { case 'A': case 'AAAA': return record.data; case 'CNAME': return record.data; case 'MX': return `${record.data} (Priority: ${record.priority || 'N/A'})`; case 'TXT': return `"${record.data}"`; case 'NS': return record.data; case 'SOA': return record.data; case 'PTR': return record.data; case 'SRV': return `${record.data} (Priority: ${record.priority || 'N/A'}, Weight: ${record.weight || 'N/A'}, Port: ${record.port || 'N/A'})`; case 'CAA': return record.data; default: return record.data; } }; const createRecordSection = (recordTypeInfo, data) => { const section = document.createElement('div'); section.className = 'record-type-section'; const header = document.createElement('h3'); header.innerHTML = `${recordTypeInfo.name} ${data.Answer ? data.Answer.length : 0}`; section.appendChild(header); if (data.Status !== 0 || !data.Answer || data.Answer.length === 0) { const noRecords = document.createElement('div'); noRecords.className = 'no-records'; noRecords.textContent = data.error ? `Error: ${data.error}` : `No ${recordTypeInfo.type} records found`; section.appendChild(noRecords); return section; } const recordList = document.createElement('ul'); recordList.className = 'record-list'; data.Answer.forEach(record => { const listItem = document.createElement('li'); const formattedData = formatRecordData(record, recordTypeInfo.type); listItem.innerHTML = `
Name: ${record.name}
TTL: ${record.TTL}s
Data: ${formattedData}
`; recordList.appendChild(listItem); }); section.appendChild(recordList); return section; }; const createSummary = (results, domain) => { const summary = document.createElement('div'); summary.className = 'summary'; const title = document.createElement('h3'); title.textContent = `DNS Summary for ${domain}`; summary.appendChild(title); const grid = document.createElement('div'); grid.className = 'summary-grid'; let totalRecords = 0; const recordCounts = {}; results.forEach((result, index) => { const recordType = RECORD_TYPES[index]; const count = (result.data.Answer && result.data.Status === 0) ? result.data.Answer.length : 0; recordCounts[recordType.type] = count; totalRecords += count; }); // Total records const totalItem = document.createElement('div'); totalItem.className = 'summary-item'; totalItem.innerHTML = `
${totalRecords}
Total Records
`; grid.appendChild(totalItem); // Individual record type counts Object.entries(recordCounts).forEach(([type, count]) => { if (count > 0) { const item = document.createElement('div'); item.className = 'summary-item'; item.innerHTML = `
${count}
${type}
`; grid.appendChild(item); } }); summary.appendChild(grid); return summary; }; const updateProgress = (current, total) => { const progressBar = document.querySelector('.progress-fill'); if (progressBar) { const percentage = (current / total) * 100; progressBar.style.width = `${percentage}%`; } }; lookupButton.addEventListener('click', async () => { const domain = domainInput.value.trim(); if (!domain) { resultsDiv.innerHTML = '

Please enter a domain name.

'; return; } // Basic domain validation const domainRegex = /^[a-zA-Z0-9][a-zA-Z0-9-]{0,61}[a-zA-Z0-9](?:\.[a-zA-Z0-9][a-zA-Z0-9-]{0,61}[a-zA-Z0-9])*$/; if (!domainRegex.test(domain)) { resultsDiv.innerHTML = '

Please enter a valid domain name.

'; return; } // Show loading state with progress bar resultsDiv.innerHTML = `

Looking up all DNS records for ${domain}...

Checking ${RECORD_TYPES.length} record types...

`; try { const results = []; // Query all record types for (let i = 0; i < RECORD_TYPES.length; i++) { const recordType = RECORD_TYPES[i]; updateProgress(i, RECORD_TYPES.length); const data = await checkDNS(domain, recordType.type); results.push({ recordType, data }); // Small delay to prevent overwhelming the DNS server await new Promise(resolve => setTimeout(resolve, 100)); } updateProgress(RECORD_TYPES.length, RECORD_TYPES.length); // Clear loading state resultsDiv.innerHTML = ''; // Add summary const summary = createSummary(results, domain); resultsDiv.appendChild(summary); // Add individual record sections results.forEach(result => { const section = createRecordSection(result.recordType, result.data); resultsDiv.appendChild(section); }); } catch (error) { resultsDiv.innerHTML = `

Error: ${error.message}

`; } }); // Allow Enter key to trigger lookup domainInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') { lookupButton.click(); } }); });
